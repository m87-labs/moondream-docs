{/* Helper functions to calculate differences */}
export const calculateDiff = (smaller, larger) => {
  const percentDiff = ((larger - smaller) / smaller * 100).toFixed(0);
  return {
    percent: percentDiff,
    reduction: ((larger - smaller) / larger * 100).toFixed(0)
  };
};

export const formatComparison = (metrics05b, metrics2b) => {
  const speedRatio = (metrics05b.speed / metrics2b.speed).toFixed(1);
  const ramDiff = calculateDiff(metrics05b.ram, metrics2b.ram);
  
  return {
    speed: {
      text: `${speedRatio}x faster`,
      comparison: `${metrics05b.speed} vs ${metrics2b.speed} tokens/s`
    },
    ram: {
      text: `${ramDiff.reduction}% less RAM`,
      comparison: `${metrics05b.ram.toLocaleString()} vs ${metrics2b.ram.toLocaleString()} MB`
    }
  };
};
